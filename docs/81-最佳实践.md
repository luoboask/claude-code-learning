## 1. 工具开发最佳实践

### 1.1 使用 buildTool() 构建器

```typescript
// ✅ 正确做法
export const MyTool = buildTool({
  name: 'MyTool',
  description: 'Tool description',
  inputSchema: z.object({
    param: z.string(),
  }),
  async call(args, context) {
    // 实现逻辑
    return { data: { result: 'success' } }
  },
  isReadOnly: () => true,
  isConcurrencySafe: () => true,
})

// ❌ 错误做法 - 直接实现 Tool 接口

  name: 'MyTool',
  // ... 手动实现所有字段，容易遗漏默认值
}
```

**优点**:
- ✅ 自动填充默认值
- ✅ 类型安全
- ✅ 集中管理默认值
- ✅ 易于维护

### 1.4 并发安全判断

```typescript
// ✅ 正确的并发安全判断
export const BashTool = buildTool({
  isConcurrencySafe(input) {
    // 读操作可并发
    return /^(ls|cat|head|tail|grep|find)\s/.test(input.command)
  },
  
  isReadOnly(input) {
    // 检查是否是只读命令
    return /^(ls|cat|head|tail|grep|find|wc)\s/.test(input.command)
  },
  
  isDestructive(input) {
    // 检查是否是破坏性命令
    return /^(rm|mv|cp|chmod|chown)\s/.test(input.command)
  },
})

// ❌ 错误做法 - 保守判断
isConcurrencySafe: () => false  // 所有工具都串行，效率低

// ❌ 错误做法 - 激进判断
isConcurrencySafe: () => true  // 所有工具都并发，可能不安全
```

## 3. 安全设计最佳实践

### 3.1 多层安全检查

```typescript
// ✅ 多层安全检查
export async function validateBashCommand(command: string): Promise<PermissionResult> {
  // 1. 命令解析
  const parsed = parseCommand(command)
  
  // 2. 基础验证
  const emptyResult = validateEmpty(parsed)
  if (emptyResult.behavior !== 'passthrough') return emptyResult
  
  // 3. 安全检查链
  const checks = [
    validateIncompleteCommands,
    validateJqSystemFunction,
    validateShellMetacharacters,
    validateDangerousVariables,
    validateCommandSubstitution,
    // ... 24 项检查
  ]
  
  for (const check of checks) {
    const result = await check(parsed)
    if (result.behavior !== 'passthrough') {
      return result  // 短路返回
    }
  }
  
  return { behavior: 'allow' }
}

// ❌ 错误做法 - 单层检查
if (!command.includes('rm')) {
  return { behavior: 'allow' }  // 绕过其他检查
}
```

### 3.3 路径验证

```typescript
// ✅ 完整的路径验证
function validateMemoryPath(raw: string | undefined, expandTilde: boolean): string | undefined {
  if (!raw) return undefined
  
  // ~/ 展开
  if (expandTilde && (candidate.startsWith('~/') || candidate.startsWith('~\\'))) {
    const rest = candidate.slice(2)
    const restNorm = normalize(rest || '.')
    if (restNorm === '.' || restNorm === '..') {
      return undefined  // 拒绝平凡剩余
    }
    candidate = join(homedir(), rest)
  }
  
  // 标准化
  const normalized = normalize(candidate).replace(/[/\\]+$/, '')
  
  // 安全检查
  if (
    !isAbsolute(normalized) ||
    normalized.length < 3 ||
    /^[A-Za-z]:$/.test(normalized) ||
    normalized.startsWith('\\\\') ||
    normalized.startsWith('//') ||
    normalized.includes('\0')
  ) {
    return undefined  // 拒绝危险路径
  }
  
  return (normalized + sep).normalize('NFC')
}
```
