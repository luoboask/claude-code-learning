## 1. 为什么需要分区？

### 1.1 问题背景

**没有分区时**:
```
工具调用 1 → 执行 → 完成
工具调用 2 → 执行 → 完成
工具调用 3 → 执行 → 完成
...
总时间 = 所有工具执行时间之和
```

**问题**: 读操作可以并发执行，但被串行处理，效率低。

### 1.2 分区解决方案

**分区后**:
```
并发批 (读操作):
  工具调用 1 ──┐
  工具调用 2 ──┼─→ 并发执行 → 完成
  工具调用 3 ──┘

串行批 (写操作):
  工具调用 4 → 执行 → 完成
  工具调用 5 → 执行 → 完成

总时间 = 并发批时间 + 串行批时间之和
```

**优点**:
- ✅ 读操作并行执行，提高效率
- ✅ 写操作串行执行，保证一致性
- ✅ 自动判断，无需手动指定

### 2.2 BashTool 示例

```typescript
// source/src/tools/BashTool/BashTool.tsx

export const BashTool = buildTool({
  // ...
  
  isConcurrencySafe(input) {
    // 读操作可并发
    return /^(ls|cat|head|tail|grep|find)\s/.test(input.command)
  },
  
  isReadOnly(input) {
    // 检查是否是只读命令
    return /^(ls|cat|head|tail|grep|find|wc)\s/.test(input.command)
  },
  
  isDestructive(input) {
    // 检查是否是破坏性命令
    return /^(rm|mv|cp|chmod|chown)\s/.test(input.command)
  },
})
```

### 2.3 默认值

```typescript
// source/src/Tool.ts

const TOOL_DEFAULTS = {
  isConcurrencySafe: (_input?: unknown) => false,  // 保守：假设不安全
  isReadOnly: (_input?: unknown) => false,         // 保守：假设写入
  isDestructive: (_input?: unknown) => false,
}
```

**原因**: 安全优先，保守默认值避免并发问题。

## 3. 分区逻辑实现

### 3.1 partitionToolCalls

```typescript
// source/src/services/tools/toolOrchestration.ts

type Batch = { isConcurrencySafe: boolean; blocks: ToolUseBlock[] }

/**
 * 将工具调用分区为并发安全批和串行批
 */
function partitionToolCalls(
  toolUseMessages: ToolUseBlock[],
  toolUseContext: ToolUseContext,
): Batch[] {
  return toolUseMessages.reduce((acc: Batch[], toolUse) => {
    // 查找工具
    const tool = findToolByName(toolUseContext.options.tools, toolUse.name)
    
    // 解析输入
    const parsedInput = tool?.inputSchema.safeParse(toolUse.input)
    
    // 判断并发安全性
    const isConcurrencySafe = parsedInput?.success
      ? (() => {
          try {
            return Boolean(tool?.isConcurrencySafe(parsedInput.data))
          } catch {
            // 如果 isConcurrencySafe 抛出异常（如 shell-quote 解析失败）
            // 保守处理：视为不安全
            return false
          }
        })()
      : false
    
    // 合并到现有批或新建批
    if (isConcurrencySafe && acc[acc.length - 1]?.isConcurrencySafe) {
      // 合并到并发批
      acc[acc.length - 1].blocks.push(toolUse)
    } else {
      // 新建串行批
      acc.push({ isConcurrencySafe, blocks: [toolUse] })
    }
    return acc
  }, [])
}
```

## 4. 并发执行

### 4.1 runToolsConcurrently

```typescript
// source/src/services/tools/toolOrchestration.ts

async function* runToolsConcurrently(
  toolUseMessages: ToolUseBlock[],
  assistantMessages: AssistantMessage[],
  canUseTool: CanUseToolFn,
  toolUseContext: ToolUseContext,
): AsyncGenerator<MessageUpdateLazy, void> {
  yield* all(
    toolUseMessages.map(async function* (toolUse) {
      // 标记为进行中
      toolUseContext.setInProgressToolUseIDs(prev =>
        new Set(prev).add(toolUse.id),
      )
      
      // 执行工具
      yield* runToolUse(
        toolUse,
        assistantMessages.find(_ =>
          _.message.content.some(
            _ => _.type === 'tool_use' && _.id === toolUse.id,
          ),
        )!,
        canUseTool,
        toolUseContext,
      )
      
      // 标记为完成
      markToolUseAsComplete(toolUseContext, toolUse.id)
    }),
    getMaxToolUseConcurrency(),  // 默认 10
  )
}
```

### 4.2 all 函数

```typescript
// source/src/utils/generators.ts

export async function* all<T>(
  generators: Array<Promise<AsyncGenerator<T>> | AsyncGenerator<T>>,
  concurrency: number = Infinity,
): AsyncGenerator<T> {
  // 并发执行多个生成器
  const pending = new Set<Promise<IteratorResult<T>>>()
  
  for (const genPromise of generators) {
    const gen = await genPromise
    
    const next = gen.next().then(result => {
      pending.delete(next)
      return result
    })
    
    pending.add(next)
    
    if (pending.size >= concurrency) {
      const { value, done } = await Promise.race(pending)
      if (!done) {
        yield value
      }
    }
  }
  
  // 等待所有完成
  for (const next of pending) {
    const { value, done } = await next
    if (!done) {
      yield value
    }
  }
}
```

## 5. 串行执行

### 5.1 runToolsSerially

```typescript
// source/src/services/tools/toolOrchestration.ts

async function* runToolsSerially(
  toolUseMessages: ToolUseBlock[],
  assistantMessages: AssistantMessage[],
  canUseTool: CanUseToolFn,
  toolUseContext: ToolUseContext,
): AsyncGenerator<MessageUpdate, void> {
  let currentContext = toolUseContext

  for (const toolUse of toolUseMessages) {
    // 标记为进行中
    toolUseContext.setInProgressToolUseIDs(prev =>
      new Set(prev).add(toolUse.id),
    )
    
    // 执行工具
    for await (const update of runToolUse(
      toolUse,
      assistantMessages.find(_ =>
        _.message.content.some(
          _ => _.type === 'tool_use' && _.id === toolUse.id,
        ),
      )!,
      canUseTool,
      currentContext,
    )) {
      if (update.contextModifier) {
        currentContext = update.contextModifier.modifyContext(currentContext)
      }
      yield {
        message: update.message,
        newContext: currentContext,
      }
    }
    
    // 标记为完成
    markToolUseAsComplete(toolUseContext, toolUse.id)
  }
}
```

## 6. 完整执行流程

### 6.1 runTools

```typescript
// source/src/services/tools/toolOrchestration.ts

export async function* runTools(
  toolUseMessages: ToolUseBlock[],
  assistantMessages: AssistantMessage[],
  canUseTool: CanUseToolFn,
  toolUseContext: ToolUseContext,
): AsyncGenerator<MessageUpdate, void> {
  let currentContext = toolUseContext
  
  // 1. 分区：并发安全批 vs 串行批
  for (const { isConcurrencySafe, blocks } of partitionToolCalls(
    toolUseMessages,
    currentContext,
  )) {
    if (isConcurrencySafe) {
      // 2. 并发执行读操作
      for await (const update of runToolsConcurrently(
        blocks,
        assistantMessages,
        canUseTool,
        currentContext,
      )) {
        yield update
      }
    } else {
      // 3. 串行执行非读操作
      for await (const update of runToolsSerially(
        blocks,
        assistantMessages,
        canUseTool,
        currentContext,
      )) {
        yield update
      }
    }
  }
}
```

### 7.3 队列处理

```typescript
// 并发执行时，上下文修改器排队处理
const queuedContextModifiers: Record<string, ((context: ToolUseContext) => ToolUseContext)[]> = {}

for await (const update of runToolsConcurrently(...)) {
  if (update.contextModifier) {
    const { toolUseID, modifyContext } = update.contextModifier
    if (!queuedContextModifiers[toolUseID]) {
      queuedContextModifiers[toolUseID] = []
    }
    queuedContextModifiers[toolUseID].push(modifyContext)
  }
  yield update
}

// 应用所有修改器
for (const block of blocks) {
  const modifiers = queuedContextModifiers[block.id]
  if (!modifiers) continue
  for (const modifier of modifiers) {
    currentContext = modifier(currentContext)
  }
}
```

## 8. 最佳实践

### 8.1 实现 isConcurrencySafe

```typescript
// ✅ 正确的实现
isConcurrencySafe(input) {
  // 读操作可并发
  return /^(ls|cat|head|tail|grep|find)\s/.test(input.command)
}

// ❌ 错误实现 - 过于激进
isConcurrencySafe() {
  return true  // 所有操作都并发，不安全
}

// ❌ 错误实现 - 过于保守
isConcurrencySafe() {
  return false  // 所有操作都串行，效率低
}
```

### 8.2 处理异常

```typescript
// ✅ 正确的异常处理
isConcurrencySafe(input) {
  try {
    // 解析命令
    const parsed = parseCommand(input.command)
    return isReadCommand(parsed)
  } catch {
    // 解析失败，保守处理
    return false
  }
}

// ❌ 错误实现 - 不处理异常
isConcurrencySafe(input) {
  const parsed = parseCommand(input.command)  // 可能抛出异常
  return isReadCommand(parsed)
}
```
