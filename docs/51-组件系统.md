// source/src/components/PromptInput/PromptInput.tsx

export function PromptInput({
  onSubmit,
  onResize,
}: PromptInputProps): React.ReactNode {
  const [input, setInput] = useState('')
  const [cursorPosition, setCursorPosition] = useState(0)
  
  useInput((inputChar, key) => {
    if (key.return) {
      onSubmit(input)
      setInput('')
    } else if (key.upArrow) {
      // 历史导航
    } else if (key.downArrow) {
      // 历史导航
    } else if (key.tab) {
      // 自动补全
    } else if (inputChar) {
      // 插入字符
      setInput(prev => insertAt(prev, cursorPosition, inputChar))
    }
  })
  
  return (
    <Box padding={1}>
      <Text>❯ </Text>
      <Text>{input}</Text>
      <Cursor position={cursorPosition} />
    </Box>
  )
}

// source/src/components/FileEditToolDiff.tsx

export function FileEditToolDiff({ diff }: FileEditDiffProps): React.ReactNode {
  return (
    <Box flexDirection="column">
      {diff.chunks.map((chunk, index) => (
        <Box key={index} flexDirection="column">
          {/* 上下文行 */}
          {chunk.lines
            .filter(line => line.type === 'context')
            .map(line => (
              <Text key={line.number} dimColor>
                {line.content}
              </Text>
            ))}
          
          {/* 删除行 */}
          {chunk.lines
            .filter(line => line.type === 'deletion')
            .map(line => (
              <Text key={line.number} backgroundColor="red">
                - {line.content}
              </Text>
            ))}
          
          {/* 添加行 */}
          {chunk.lines
            .filter(line => line.type === 'addition')
            .map(line => (
              <Text key={line.number} backgroundColor="green">
                + {line.content}
              </Text>
            ))}
        </Box>
      ))}
    </Box>
  )
}

### 4.3 Spinner 组件

```typescript
// source/src/components/Spinner.tsx

export function Spinner({ mode = 'dots' }: SpinnerProps): React.ReactNode {
  const [frame, setFrame] = useState(0)
  
  useEffect(() => {
    const interval = setInterval(() => {
      setFrame(prev => (prev + 1) % FRAMES[mode].length)
    }, 80)
    
    return () => clearInterval(interval)
  }, [mode])
  
  return <Text>{FRAMES[mode][frame]}</Text>
}

const FRAMES = {
  dots: ['⠋', '⠙', '⠹', '⠸', '⠼', '⠴', '⠦', '⠧', '⠇', '⠏'],
  line: ['-', '\\', '|', '/'],
  triangle: ['◢', '◣', '◤', '◥'],
}
```

## 5. 权限组件

### 5.1 PermissionDialog 组件

```typescript
// source/src/components/permissions/PermissionDialog.tsx

export function PermissionDialog({
  tool,
  input,
  onAllow,
  onDeny,
}: PermissionDialogProps): React.ReactNode {
  const [selected, setSelected] = useState<'allow' | 'deny'>('allow')
  const [permanent, setPermanent] = useState(false)
  
  useInput((input, key) => {
    if (key.leftArrow || key.rightArrow) {
      setSelected(prev => prev === 'allow' ? 'deny' : 'allow')
    } else if (key.return) {
      if (selected === 'allow') {
        onAllow(permanent)
      } else {
        onDeny(permanent)
      }
    }
  })
  
  return (
    <Box flexDirection="column" border={true} padding={1}>
      <Text bold>Permission Request</Text>
      <Text>{tool.name}: {input.command}</Text>
      
      <Box marginTop={1}>
        <Text color={selected === 'allow' ? 'green' : undefined}>
          [A] Allow
        </Text>
        <Text marginX={2}>|</Text>
        <Text color={selected === 'deny' ? 'red' : undefined}>
          [D] Deny
        </Text>
      </Box>
      
      <Box marginTop={1}>
        <Text>
          [{permanent ? 'x' : ' '}] Always {selected} for this tool
        </Text>
      </Box>
    </Box>
  )
}
```

## 8. 最佳实践

### 8.1 组件设计

```markdown
✅ 应该做:

1. 单一职责
   - 每个组件只做一件事
   - 保持小组件可组合

2. 清晰的 Props 接口
   - 使用 TypeScript 类型定义
   - 提供默认值

3. 可复用性
   - 提取通用组件
   - 使用设计系统组件

❌ 不应该做:

1. 大组件
2. 隐式依赖
3. 直接操作 DOM
```

### 8.2 性能优化

```markdown
✅ 应该做:

1. 使用 React.memo
2. 避免不必要的重渲染
3. 虚拟长列表

❌ 不应该做:

1. 在渲染中创建对象
2. 传递新函数作为 props
3. 渲染过多组件
```
