```python
from typing import Generic, TypeVar, Optional
from pydantic import BaseModel

InputT = TypeVar('InputT', bound=BaseModel)
OutputT = TypeVar('OutputT')

class Tool(Generic[InputT, OutputT]):
    # 基础属性
    name: str
    description: str
    input_schema: type[InputT]
    
    # 核心方法
    async def call(self, args: InputT, context: ToolContext) -> ToolResult[OutputT]:
        pass
    
    # 特性检查
    def is_enabled(self) -> bool:
        return True
    
    def is_concurrency_safe(self, args: InputT) -> bool:
        return False
    
    def is_read_only(self, args: InputT) -> bool:
        return False
    
    # 权限检查
    async def check_permissions(self, args: InputT, context: ToolContext) -> PermissionResult:
        return PermissionResult(behavior='allow')
```

## 2. 构建器模式对比

### Claude Code

```typescript
const TOOL_DEFAULTS = {
  isEnabled: () => true,
  isConcurrencySafe: () => false,  // 保守
  isReadOnly: () => false,         // 保守
  isDestructive: () => false,
  checkPermissions: (input) => 
    Promise.resolve({ behavior: 'allow', updatedInput: input }),
}

export function buildTool<D>(def: D): BuiltTool<D> {
  return {
    ...TOOL_DEFAULTS,
    userFacingName: () => def.name,
    ...def,
  } as BuiltTool<D>
}
```

### evo-agents 现状

```python
# 无构建器模式
class MyTool(BaseTool):
    def __init__(self):
        self.name = "MyTool"
        # 手动设置所有属性
```

### 借鉴建议

```python
# ✅ 建议添加
TOOL_DEFAULTS = {
    'is_enabled': lambda: True,
    'is_concurrency_safe': lambda args: False,
    'is_read_only': lambda args: False,
    'is_destructive': lambda args: False,
    'check_permissions': lambda args, ctx: PermissionResult(behavior='allow'),
}

def build_tool(**kwargs):
    """构建工具，填充默认值"""
    tool = type('Tool', (BaseTool,), kwargs)
    
    # 填充默认值
    for key, default in TOOL_DEFAULTS.items():
        if not hasattr(tool, key):
            setattr(tool, key, default)
    
    return tool
```

## 3. 权限系统对比

### Claude Code

```
三层规则:
- alwaysAllowRules  → 总是允许
- alwaysDenyRules   → 总是拒绝
- alwaysAskRules    → 总是询问

四种模式:
- default  → 默认模式
- bypass   → 绕过模式
- auto     → 自动模式
- plan     → 计划模式
```

### evo-agents 现状

```python
# 简单权限检查
if tool.requires_permission:
    ask_user()
else:
    execute()
```

### 借鉴建议

```python
# ✅ 建议添加
class PermissionSystem:
    def __init__(self):
        self.always_allow_rules = {}
        self.always_deny_rules = {}
        self.always_ask_rules = {}
        self.mode = 'default'
    
    async def check_permission(self, tool: Tool, args: dict) -> PermissionResult:
        # 1. 检查 alwaysAllowRules
        if self.match_rule(self.always_allow_rules, tool, args):
            return PermissionResult(behavior='allow')
        
        # 2. 检查 alwaysDenyRules
        if self.match_rule(self.always_deny_rules, tool, args):
            return PermissionResult(behavior='deny')
        
        # 3. 检查 alwaysAskRules
        if self.match_rule(self.always_ask_rules, tool, args):
            return PermissionResult(behavior='ask')
        
        # 4. 检查模式
        if self.mode == 'bypass' and not tool.is_destructive(args):
            return PermissionResult(behavior='allow')
        
        # 5. 默认询问
        return PermissionResult(behavior='ask')
```

## 4. 执行编排对比

### Claude Code

```typescript
// 分区：并发安全批 vs 串行批
for (const { isConcurrencySafe, blocks } of partitionToolCalls(...)) {
  if (isConcurrencySafe) {
    // 并发执行读操作
    for await (const update of runToolsConcurrently(...)) {
      yield update
    }
  } else {
    // 串行执行非读操作
    for await (const update of runToolsSerially(...)) {
      yield update
    }
  }
}
```

### evo-agents 现状

```python

    result = await execute_tool(tool_call)
```

### 借鉴建议

```python
# ✅ 建议添加
async def run_tools(tool_calls: list, context: ToolContext):
    """执行工具，支持并发/串行分区"""
    
    # 分区：并发安全批 vs 串行批
    batches = partition_tool_calls(tool_calls, context)
    
    for batch in batches:
        if batch.is_concurrency_safe:
            # 并发执行读操作
            results = await asyncio.gather(
                *[execute_tool(call, context) for call in batch.calls]
            )
            yield from results
        else:
            # 串行执行非读操作
            for call in batch.calls:
                result = await execute_tool(call, context)
                yield result

def partition_tool_calls(tool_calls: list, context: ToolContext) -> list:
    """分区工具调用"""
    batches = []
    current_batch = None
    
    for call in tool_calls:
        tool = find_tool_by_name(call.name)
        is_safe = tool.is_concurrency_safe(call.args)
        
        if is_safe and current_batch and current_batch.is_safe:
            # 合并到并发批
            current_batch.calls.append(call)
        else:
            # 新建批
            current_batch = Batch(is_safe=is_safe, calls=[call])
            batches.append(current_batch)
    
    return batches
```

### 借鉴建议

```python
# ✅ 建议增强
class MCPClient:
    def __init__(self, server_name: str, config: dict):
        self.server_name = server_name
        self.transport = self.create_transport(config)
        self.client = Client(name='evo-agents')
    
    def create_transport(self, config: dict) -> Transport:
        """创建传输层"""
        if config['type'] == 'stdio':
            return StdioTransport(config['command'], config['args'])
        elif config['type'] == 'sse':
            return SSETransport(config['url'])
        elif config['type'] == 'websocket':
            return WebSocketTransport(config['url'])
    
    async def connect(self):
        """连接 MCP 服务器"""
        await self.client.connect(self.transport)
    
    async def list_tools(self) -> list:
        """列出工具"""
        return await self.client.request('tools/list')
    
    async def call_tool(self, name: str, args: dict) -> ToolResult:
        """调用工具"""
        result = await self.client.request('tools/call', {
            'name': name,
            'arguments': args
        })
        if result.get('isError'):
            raise MCPToolError(result['content'])
        return ToolResult(data=result['content'])
```
