## 1. 最小状态管理原语

### 1.1 createStore 实现

```typescript
// source/src/state/store.ts (836B)

type Listener = () => void
type OnChange<T> = (args: { newState: T; oldState: T }) => void

export type Store<T> = {
  getState: () => T
  setState: (updater: (prev: T) => T) => void
  subscribe: (listener: Listener) => () => void
}

export function createStore<T>(
  initialState: T,
  onChange?: OnChange<T>,
): Store<T> {
  let state = initialState
  const listeners = new Set<Listener>()

  return {
    getState: () => state,

    setState: (updater: (prev: T) => T) => {
      const prev = state
      const next = updater(prev)
      
      // 无变更不通知
      if (Object.is(next, prev)) return
      
      state = next
      
      // 通知变更处理器
      onChange?.({ newState: next, oldState: prev })
      
      // 通知所有订阅者
      for (const listener of listeners) listener()
    },

    subscribe: (listener: Listener) => {
      listeners.add(listener)
      return () => listeners.delete(listener)
    },
  }
}
```

### 1.2 核心特性

| 特性 | 实现 | 说明 |
|------|------|------|
| **不可变更新** | `updater(prev) => next` | 返回新对象，不修改原状态 |
| **无变更优化** | `Object.is(next, prev)` | 状态未变时不通知 |
| **发布订阅** | `listeners.forEach(fn => fn())` | 通知所有订阅者 |
| **变更追踪** | `onChange?.({ newState, oldState })` | 可选的变更处理器 |

### 1.3 使用示例

```typescript
// 1. 创建 store
const store = createStore({ count: 0, todos: [] })

// 2. 订阅变更
const unsubscribe = store.subscribe(() => {
  console.log('状态变更:', store.getState())
})

// 3. 更新状态（不可变模式）
store.setState(prev => ({
  ...prev,
  count: prev.count + 1,
}))

// 4. 取消订阅
unsubscribe()
```

## 2. 不可变更新模式

### 2.1 正确做法

```typescript
// ✅ 不可变更新
store.setState(prev => ({
  ...prev,
  tasks: {
    ...prev.tasks,
    [taskId]: { ...prev.tasks[taskId], status: 'completed' }
  }
}))

// ✅ 数组更新
store.setState(prev => ({
  ...prev,
  todos: [...prev.todos, newTodo]
}))

// ✅ 深层嵌套更新
store.setState(prev => ({
  ...prev,
  mcp: {
    ...prev.mcp,
    clients: prev.mcp.clients.map(client =>
      client.id === clientId ? { ...client, status: 'connected' } : client
    )
  }
}))
```

### 2.2 错误做法

```typescript
// ❌ 可变更新（直接修改）
state.count = 1
state.todos.push(newTodo)
state.tasks[taskId].status = 'completed'

// ❌ 浅拷贝但修改了嵌套对象
store.setState(prev => ({
  ...prev,
  tasks: prev.tasks,  // 引用相同，未创建新对象
}))
```

### 2.3 为什么使用不可变？

| 优点 | 说明 |
|------|------|
| **可预测** | 状态变更清晰，易于追踪 |
| **易于调试** | 可以对比前后状态 |
| **支持时间旅行** | 可以保存历史状态快照 |
| **避免副作用** | 不会意外修改共享数据 |
| **React 友好** | React 依赖不可变检测变更 |

### 4.1 完整流程

```
工具执行/用户操作
    │
    ↓
┌─────────────────────────────────┐
│ AppStateStore.setState()        │
│ (不可变更新：prev => ({...}))   │
└───────────────┬─────────────────┘

                ↓
┌─────────────────────────────────┐
│ onChangeAppState()              │
│ - 持久化 Settings               │
│ - 发送遥测                      │
│ - 触发副作用 (MCP 重连等)        │
└───────────────┬─────────────────┘
                │
                ↓
┌─────────────────────────────────┐
│ Store 通知所有订阅者             │

└───────────────┬─────────────────┘
                │
                ↓
┌─────────────────────────────────┐
│ Ink 组件重新渲染                 │

└─────────────────────────────────┘
```

### 4.2 onChangeAppState 实现

```typescript
// source/src/state/onChangeAppState.ts

export function onChangeAppState({ newState, oldState }: {
  newState: AppState
  oldState: AppState
}) {
  // 1. 持久化 Settings
  if (newState.settings !== oldState.settings) {
    saveGlobalConfig(newState.settings)
  }
  
  // 2. 发送遥测
  if (newState.mcp.clients !== oldState.mcp.clients) {
    logEvent('mcp_clients_changed', {
      count: newState.mcp.clients.length,
    })
  }
  
  // 3. 触发副作用
  if (newState.replBridgeEnabled !== oldState.replBridgeEnabled) {
    // Bridge 重连
    reconnectBridge()
  }
  
  if (newState.mcp.pluginReconnectKey !== oldState.mcp.pluginReconnectKey) {
    // MCP 插件重连
    reconnectMcpPlugins()
  }
}
```

```typescript
// source/src/hooks/useCanUseTool.tsx

export type CanUseToolFn<Input> = (
  tool: Tool,
  input: Input,
  toolUseContext: ToolUseContext,
  assistantMessage: AssistantMessage,
  toolUseID: string,
  forceDecision?: PermissionDecision<Input>,
) => Promise<PermissionDecision<Input>>

function useCanUseTool(setToolUseConfirmQueue, setToolPermissionContext) {
  return async (tool, input, toolUseContext, assistantMessage, toolUseID, forceDecision) => 
    new Promise(resolve => {
      // 1. 创建权限上下文
      const ctx = createPermissionContext(tool, input, toolUseContext, ...)
      
      // 2. 检查是否已中止
      if (ctx.resolveIfAborted(resolve)) return
      
      // 3. 获取权限决策
      const decisionPromise = forceDecision !== undefined
        ? Promise.resolve(forceDecision)
        : hasPermissionsToUseTool(tool, input, toolUseContext, assistantMessage, toolUseID)
      
      decisionPromise.then(async result => {
        // 4. 处理决策
        if (result.behavior === 'allow') {
          ctx.logDecision({ decision: 'accept', source: 'config' })
          resolve(ctx.buildAllow(result.updatedInput ?? input))
          return
        }
        
        if (result.behavior === 'deny') {
          logPermissionDecision({...}, { decision: 'reject', source: 'config' })
          resolve(result)
          return
        }
        
        if (result.behavior === 'ask') {
          // 交互式权限对话框
          handleInteractivePermission({...}, resolve)
          return
        }
      })
    })
}
```

### 5.3 useSettings 示例

```typescript
// source/src/hooks/useSettings.ts

function useSettings() {
  const [settings, setSettings] = useState<AppState['settings']>()
  
  useEffect(() => {
    // 订阅状态变更
    const unsubscribe = store.subscribe(() => {
      setSettings(store.getState().settings)
    })
    
    // 初始读取
    setSettings(store.getState().settings)
    
    return unsubscribe
  }, [])
  
  const updateSetting = useCallback((key: string, value: unknown) => {
    store.setState(prev => ({
      ...prev,
      settings: {
        ...prev.settings,
        [key]: value,
      }
    }))
  }, [])
  
  return { settings, updateSetting }
}
```

### 7.2 选择性订阅

```typescript
// 只订阅关心的状态
function useSelectedState<T>(selector: (state: AppState) => T): T {
  const [value, setValue] = useState(() => selector(store.getState()))
  
  useEffect(() => {
    const unsubscribe = store.subscribe(() => {
      const newValue = selector(store.getState())
      if (!Object.is(newValue, value)) {
        setValue(newValue)
      }
    })
    return unsubscribe
  }, [selector, value])
  
  return value
}

// 使用
const taskCount = useSelectedState(state => Object.keys(state.tasks).length)
```

## 9. 最佳实践

### 9.1 状态设计

```markdown
✅ 应该做:
1. 使用不可变更新
2. 扁平化状态结构
3. 按功能组织状态
4. 添加状态检查
5. 使用选择性订阅

❌ 不应该做:
1. 直接修改状态
2. 深层嵌套状态
3. 在状态中存储函数
4. 过度使用全局状态
5. 不必要的状态复制
```
