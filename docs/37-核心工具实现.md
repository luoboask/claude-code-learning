```typescript
export const BashTool = buildTool({

    command: z.string().describe('The command to execute'),

    // 实现见下方
```

### 1.2 安全检查流程

```typescript
async call(args, context) {
  // 1. 命令解析
  const parsed = parseCommand(args.command)
  
  // 2. 安全验证 (24 项检查)
  const checks = [
    validateEmpty,                    // 空命令检查
    validateIncompleteCommands,       // 不完整命令
    validateJqSystemFunction,         // jq system() 函数
    validateShellMetacharacters,      // Shell 元字符
    validateDangerousVariables,       // 危险变量
    validateCommandSubstitution,      // 命令替换
    validateInputRedirection,         // 输入重定向
    validateOutputRedirection,        // 输出重定向
    validateIfsInjection,             // IFS 注入
    validateGitCommitSubstitution,    // Git commit 替换
    validateProcEnvironAccess,        // /proc 访问
    validateMalformedTokenInjection,  // Token 注入
    validateBackslashEscapedWhitespace, // 反斜杠转义
    validateBraceExpansion,           // 大括号展开
    validateControlCharacters,        // 控制字符
    validateUnicodeWhitespace,        // Unicode 空白
    validateMidWordHash,              // 词中#
    validateZshDangerousCommands,     // Zsh 危险命令
    validateBackslashEscapedOperators, // 转义操作符
    validateCommentQuoteDesync,       // 注释引号失配
    validateQuotedNewline,            // 引号内换行
  ]
  
  for (const check of checks) {
    const result = await check(parsed)
    if (result.behavior !== 'passthrough') {
      return { data: { error: result.message } }
    }
  }
  
  // 3. 权限检查
  const permissionResult = await canUseTool(this, args, context, parentMessage)
  if (permissionResult.behavior !== 'allow') {
    return { data: { error: 'Permission denied' } }
  }
  
  // 4. 执行命令
  const result = await exec(args.command, {
    timeout: args.timeout,
    cwd: context.options.cwd,
  })
  
  // 5. 返回结果
  return {
    data: {
      stdout: result.stdout,
      stderr: result.stderr,
      exitCode: result.exitCode,
    },
  }
}
```

```typescript
// 1. Zsh 危险命令检测
function validateZshDangerousCommands(parsed): PermissionResult {
  const ZSH_DANGEROUS_COMMANDS = new Set([
    'zmodload',    // 模块加载（危险网关）
    'emulate',     // 模拟模式（eval 等价）
    'sysopen',     // 文件描述符操作
    'sysread',     // 文件描述符读取
    'syswrite',    // 文件描述符写入
    'zpty',        // 伪终端执行
    'ztcp',        // TCP 连接（数据外泄）
    'zf_rm',       // 内置 rm（绕过检查）
    'zf_mv',       // 内置 mv
    'zf_ln',       // 内置 ln
    'zf_chmod',    // 内置 chmod
    'zf_chown',    // 内置 chown
  ])
  
  if (ZSH_DANGEROUS_COMMANDS.has(parsed.baseCommand)) {
    return {
      behavior: 'ask',
      message: `Command uses Zsh dangerous command: ${parsed.baseCommand}`,
    }
  }
  
  return { behavior: 'passthrough' }
}

// 2. 命令替换检测
function validateCommandSubstitution(parsed): PermissionResult {
  const COMMAND_SUBSTITUTION_PATTERNS = [
    { pattern: /<\(/, message: 'process substitution <()' },
    { pattern: />\(/, message: 'process substitution >()' },
    { pattern: /\$\(/, message: '$() command substitution' },
    { pattern: /\$\{/, message: '${} parameter substitution' },
    { pattern: /(?:^|[\s;&|])=[a-zA-Z_]/, message: 'Zsh equals expansion' },
  ]
  
  for (const { pattern, message } of COMMAND_SUBSTITUTION_PATTERNS) {
    if (pattern.test(parsed.unquotedContent)) {
      return {
        behavior: 'ask',
        message: `Command contains dangerous pattern: ${message}`,
      }
    }
  }
  
  return { behavior: 'passthrough' }
}

// 3. Heredoc 安全检测
function validateSafeHeredoc(command): PermissionResult {
  const HEREDOC_IN_SUBSTITUTION = /\$\(.*<</
  
  if (!HEREDOC_IN_SUBSTITUTION.test(command)) {
    return { behavior: 'passthrough' }
  }
  
  // 检查是否是安全的 heredoc 模式
  if (isSafeHeredoc(command)) {
    return {
      behavior: 'allow',
      decisionReason: { type: 'other', reason: 'Safe heredoc substitution' },
    }
  }
  
  return { behavior: 'passthrough' }
}
```

### 2.2 Diff 应用

```typescript
// source/src/tools/FileEditTool/utils.ts

export function applyDiff(
  content: string,
  oldString: string,
  newString: string,
): string {
  // 1. 查找 old_string
  const index = content.indexOf(oldString)
  
  if (index === -1) {
    throw new Error('old_string not found in file')
  }
  
  // 2. 检查唯一性
  const lastIndex = content.lastIndexOf(oldString)
  if (index !== lastIndex) {
    throw new Error('old_string is not unique in file')
  }
  
  // 3. 应用替换
  return content.slice(0, index) + newString + content.slice(index + oldString.length)
}

// 带上下文的 diff
export function applyDiffWithContext(
  content: string,
  oldString: string,
  newString: string,
  context: number = 3,
): string {
  const lines = content.split('\n')
  const oldLines = oldString.split('\n')
  
  // 查找匹配位置（带上下文）
  for (let i = 0; i < lines.length - oldLines.length + 1; i++) {
    const slice = lines.slice(i, i + oldLines.length)
    
    if (slice.join('\n') === oldString) {
      // 找到匹配，应用替换
      const newLines = newString.split('\n')
      return [
        ...lines.slice(0, i),
        ...newLines,
        ...lines.slice(i + oldLines.length),
      ].join('\n')
    }
  }
  
  throw new Error('old_string not found in file')
}
```

```typescript
// source/src/tools/AgentTool/forkSubagent.ts

export async function runForkedAgent(params: {
  prompt: string
  context: ToolUseContext
  worktreeCwd: string
}): Promise<{ output: string; output_file: string }> {
  // 1. 创建工作树（隔离的文件系统）
  const worktree = await createWorktree(params.context.options.cwd)
  
  // 2. 构建子代理提示
  const systemPrompt = buildSubagentSystemPrompt({
    parentCwd: params.context.options.cwd,
    worktreeCwd: worktree,
  })
  
  // 3. 运行子代理
  const result = await runAgent({
    prompt: params.prompt,
    systemPrompt,
    cwd: worktree,
    tools: getToolsForSubagent(),
  })
  
  return {
    output: result.output,
    output_file: join(worktree, 'output.txt'),
  }
}
```

```typescript
// source/src/tools/MCPTool/MCPTool.ts

export const MCPTool = buildTool({
  name: 'mcp__server__tool',
  description: 'Call MCP tool from external server',
  
  isMcp: true,
  mcpInfo: {
    serverName: 'server',
    toolName: 'tool',
  },
  
  async call(args, context) {
    // 1. 获取 MCP 客户端
    const client = context.options.mcpClients.find(

    )
    
    if (!client) {
      throw new Error(`MCP server not found: ${args.server}`)

    // 2. 调用工具
    const result = await client.callTool(args.tool, args.args)
    
    // 3. 返回结果
    return {
      data: result.content,
      mcpMeta: result._meta,
    }
  },
})
```

```typescript
// source/src/services/mcp/client.ts

export class McpClient {
  private client: Client
  private transport: Transport
  
  constructor(serverName: string, config: McpServerConfig) {
    if (config.type === 'stdio') {
      this.transport = new StdioClientTransport({
        command: config.command,
        args: config.args,
      })
    } else if (config.type === 'sse') {
      this.transport = new SSEClientTransport(new URL(config.url))
    } else if (config.type === 'websocket') {
      this.transport = new WebSocketTransport(config.url)
    }
    
    this.client = new Client({
      name: 'claude-code',
      version: pkg.version,
    })
  }
  
  async connect(): Promise<void> {
    await this.client.connect(this.transport)
  }
  
  async listTools(): Promise<ListToolsResult> {
    return this.client.request({ method: 'tools/list' }, ListToolsResultSchema)
  }
  
  async callTool(name: string, args: Record<string, unknown>): Promise<MCPToolResult> {
    const result = await this.client.request(
      { method: 'tools/call', params: { name, arguments: args } },
      CallToolResultSchema,
    )
    
    if (result.isError) {
      throw new McpToolCallError(result.content?.[0]?.text ?? 'Unknown error')
    }
    
    return {
      content: result.content,
      _meta: result._meta,
    }
  }
}
```

```typescript
// ✅ 完整的错误处理
async call(args, context) {
  try {
    // 1. 验证输入
    const validation = await this.validateInput?.(args, context)
    if (validation && !validation.result) {
      return { data: { error: validation.message } }
    }
    
    // 2. 执行逻辑
    const result = await execute(args)
    
    return { data: result }
  } catch (error) {
    return {
      data: {
        error: error instanceof Error ? error.message : 'Unknown error',
      },
    }
  }
}

// ❌ 不完整的错误处理
async call(args, context) {

}
```
